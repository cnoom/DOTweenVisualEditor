using System;
using UnityEngine;
using UnityEngine.Events;
using DG.Tweening;

namespace CNoom.DOTweenVisual.Data
{
    [Serializable]
    public class TweenStepData
    {
        // 基本信息
        public string StepName = "New Step";
        public bool IsEnabled = true;
        
        // 动画类型
        public TweenType Type = TweenType.Move;
        
        // 时长控制
        public float Duration = 1f;
        public float Delay = 0f;
        
        // 缓动曲线
        public Ease Ease = Ease.OutQuad;
        public bool UseCustomCurve = false;
        public AnimationCurve CustomCurve;
        
        // Transform 相关参数
        public TransformTarget TransformTarget = TransformTarget.Position;
        public Vector3 TargetValue = Vector3.zero;
        public bool IsRelative = false;
        
        // 执行模式
        public ExecutionMode ExecutionMode = ExecutionMode.Append;
        public float InsertTime = 0f;
        
        // 回调
        public UnityEvent OnComplete;
        
        // 属性动画（预留）
        public string PropertyName = "";
        public float PropertyValueFrom = 0f;
        public float PropertyValueTo = 1f;
    }
}

## 三、主组件设计

### DOTweenVisualPlayer

```csharp
using System.Collections.Generic;
using UnityEngine;
using DG.Tweening;
using CNoom.DOTweenVisual.Data;
using CNoom.DOTweenVisual.Adapter;
using CNoom.DOTweenVisual.Pool;

namespace CNoom.DOTweenVisual.Components
{
    public class DOTweenVisualPlayer : MonoBehaviour
    {
        [Header("动画序列")]
        [SerializeField] private List<TweenStepData> _steps = new();
        
        [Header("播放设置")]
        [SerializeField] private bool _playOnStart = false;
        [SerializeField] private int _loops = 1;
        [SerializeField] private LoopType _loopType = LoopType.Restart;
        
        [Header("调试")]
        [SerializeField] private bool _debugMode = false;
        
        private Sequence _currentSequence;
        private bool _isPlaying;
        
        public IReadOnlyList<TweenStepData> Steps => _steps;
        public bool IsPlaying => _isPlaying;
        public int StepCount => _steps.Count;
        
        #region Public API
        
        public void Play()
        {
            if (_isPlaying) return;
            BuildAndPlay();
        }
        
        public void Stop()
        {
            KillSequence();
        }
        
        public void Pause()
        {
            _currentSequence?.Pause();
        }
        
        public void Restart()
        {
            Stop();
            Play();
        }
        
        #endregion
        
        #region Editor API
        
        public void AddStep(TweenStepData step)
        {
            _steps.Add(step);
        }
        
        public void RemoveStep(int index)
        {
            if (index >= 0 && index < _steps.Count)
                _steps.RemoveAt(index);
        }
        
        public void MoveStep(int fromIndex, int toIndex)
        {
            if (fromIndex < 0 || fromIndex >= _steps.Count) return;
            if (toIndex < 0 || toIndex >= _steps.Count) return;
            
            var step = _steps[fromIndex];
            _steps.RemoveAt(fromIndex);
            _steps.Insert(toIndex, step);
        }
        
        public void ClearAllSteps()
        {
            _steps.Clear();
        }
        
        #endregion
        
        #region Private Methods
        
        private void Start()
        {
            if (_playOnStart)
                Play();
        }
        
        private void OnDestroy()
        {
            KillSequence();
        }
        
        private void BuildAndPlay()
        {
            KillSequence();
            _currentSequence = DOTween.Sequence();
            
            foreach (var step in _steps)
            {
                if (!step.IsEnabled) continue;
                AppendStepToSequence(step);
            }
            
            _currentSequence.SetLoops(_loops, _loopType);
            _currentSequence.Play();
            _isPlaying = true;
        }
        
        private void AppendStepToSequence(TweenStepData step)
        {
            // TODO: 根据类型创建 Tween 并添加到序列
        }
        
        private void KillSequence()
        {
            if (_currentSequence != null)
            {
                _currentSequence.Kill();
                _currentSequence = null;
            }
            _isPlaying = false;
        }
        
        #endregion
    }
}
```
