## 三、数据结构

### TweenStepData

```csharp
[Serializable]
public class TweenStepData
{
    public string StepName = "New Step";
    public TweenStepType Type;          // Move/Rotate/Scale/Property/Callback
    public float Duration = 1f;
    public float Delay = 0f;
    public Ease Ease = Ease.OutQuad;
    public AnimationCurve CustomCurve;
    
    // Transform 相关
    public TransformTarget TransformTarget;
    public Vector3 TargetValue;
    public bool IsRelative = false;
    
    // 执行模式
    public ExecutionMode ExecutionMode = ExecutionMode.Append;
    public float InsertTime = 0f;       // Insert 模式时使用
    
    // 回调
    public UnityEvent Callback;
}

public enum TweenStepType
{
    Move,
    Rotate,
    Scale,
    Property,
    Delay,
    Callback
}

public enum TransformTarget
{
    Position,
    Rotation,
    Scale
}

public enum ExecutionMode
{
    Append,   // 顺序执行
    Join,     // 并行执行
    Insert    // 指定时间插入
}
```

### DOTweenVisualPlayer

```csharp
public class DOTweenVisualPlayer : MonoBehaviour
{
    [SerializeField] private List<TweenStepData> _steps = new();
    [SerializeField] private bool _playOnStart = false;
    [SerializeField] private int _loops = 1;
    [SerializeField] private LoopType _loopType = LoopType.Restart;
    
    public IReadOnlyList<TweenStepData> Steps => _steps;
    
    public void Play();
    public void Stop();
    public void Pause();
    public void Restart();
    
    // 编辑器专用
    public void AddStep(TweenStepData step);
    public void RemoveStep(int index);
    public void MoveStep(int fromIndex, int toIndex);
}
```

## 四、对象池设计

### TweenPool（全局单例）

```csharp
public class TweenPool : MonoBehaviour
{
    public static TweenPool Instance { get; }
    
    public T Get<T>(Func<T> creator) where T : Tween;
    public void Return(Tween tween);
    public void Clear();
}
```

**工作流程**：
1. 播放时从池中获取 Tweener
2. 播放完成后回收入池
3. 避免频繁 GC 分配
